1. Introduction to JavaScript

What is JavaScript?

JavaScript is a high-level, interpreted programming language that is one of the core technologies of the World Wide Web. It enables interactive web pages and is an essential part of web applications.

Key Features:

  • Client-side Scripting: Runs in the browser without server interaction
  • Dynamic Typing: Variables don't need type declarations
  • Event-driven: Responds to user interactions
  • Object-oriented: Supports OOP with prototypes
  • Functional: Functions are first-class citizens

2. Variables & Data Types

What are Variables?

Variables are containers that store data values. In JavaScript, you can declare variables using var, let, or const.

Variable Declaration

// Using var (function-scoped, can be redeclared)
var name = "John";
var age = 25;

// Using let (block-scoped, cannot be redeclared)
let city = "New York";
let count = 0;

// Using const (block-scoped, cannot be reassigned)
const PI = 3.14159;
const MAX_SIZE = 100;

// Multiple variables
let x = 5, y = 10, z = 15;

Difference Between var, let, and const

Keyword Scope Redeclare Reassign
var Function-scoped Yes Yes
let Block-scoped No Yes
const Block-scoped No No

Data Types in JavaScript

JavaScript has 8 data types:

Primitive Types

// String
let name = "John";
let greeting = 'Hello';

// Number
let age = 25;
let price = 99.99;
let temperature = -10;

// Boolean
let isStudent = true;
let isActive = false;

// Undefined
let value;
console.log(value); // undefined

// Null
let empty = null;

// BigInt (for large integers)
let bigNumber = 9007199254740991n;

// Symbol (unique identifier)
let id = Symbol('id');

Object Type

// Object
let person = {
    name: "John",
    age: 25,
    city: "New York"
};

// Array
let fruits = ["apple", "banana", "cherry"];

// Function
function greet() {
    return "Hello";
}

Type Checking

// typeof operator
console.log(typeof "Hello");     // "string"

console.log(typeof true);         // "boolean"
console.log(typeof undefined);    // "undefined"
console.log(typeof null);         // "object" (this is a bug in JavaScript)

Type Conversion

// String to Number

3. Operators

What are Operators?

Operators are symbols that perform operations on values and variables. JavaScript has various types of operators.

Arithmetic Operators

let a = 10;
let b = 3;

console.log(a + b);  // 13 (Addition)
console.log(a - b);  // 7 (Subtraction)
console.log(a * b);  // 30 (Multiplication)
console.log(a / b);  // 3.333... (Division)
console.log(a % b);  // 1 (Modulus - remainder)
console.log(a ** b); // 1000 (Exponentiation)
console.log(++a);    // 11 (Pre-increment)
console.log(a++);    // 11 (Post-increment)
console.log(--b);    // 2 (Pre-decrement)
console.log(b--);    // 2 (Post-decrement)

Assignment Operators

let x = 10;

x += 5;  // x = x + 5 (15)
x -= 3;  // x = x - 3 (12)
x *= 2;  // x = x * 2 (24)
x /= 4;  // x = x / 4 (6)
x %= 4;  // x = x % 4 (2)
x **= 3; // x = x ** 3 (8)

Comparison Operators

let a = 5;
let b = 10;

console.log(a == b);   // false (loose equality)
console.log(a != b);   // true (loose inequality)
console.log(a === b);  // false (strict equality)
console.log(a !== b);  // true (strict inequality)
console.log(a < b);    // true (less than)
console.log(a > b);    // false (greater than)
console.log(a <= b);   // true (less than or equal)
console.log(a >= b);   // false (greater than or equal)

// Loose vs Strict Equality
console.log(5 == "5");   // true (type coercion)
console.log(5 === "5");  // false (no type coercion)

Logical Operators

let x = true;
let y = false;

console.log(x && y);   // false (AND)
console.log(x || y);   // true (OR)
console.log(!x);       // false (NOT)

// Short-circuit evaluation
let result = true && "Hello";  // "Hello"
let result2 = false || "World"; // "World"

// Nullish coalescing (??)
let value = null ?? "default";  // "default"
let value2 = undefined ?? "default"; // "default"
let value3 = 0 ?? "default";     // 0

Ternary Operator

// Syntax: condition ? valueIfTrue : valueIfFalse
let age = 18;
let status = age >= 18 ? "Adult" : "Minor";
console.log(status); // "Adult"

// Nested ternary
let score = 85;
let grade = score >= 90 ? "A" : 
            score >= 80 ? "B" : 
            score >= 70 ? "C" : "F";
console.log(grade); // "B"

Type Operators

// typeof
console.log(typeof "Hello");  // "string"
console.log(typeof 42);        // "number"

// instanceof
let arr = [1, 2, 3];
console.log(arr instanceof Array);  // true
console.log(arr instanceof Object);  // true

String Operators

// Concatenation
let firstName = "John";
let lastName = "Doe";
let fullName = firstName + " " + lastName; // "John Doe"

// Template literals
let message = `Hello, ${firstName}!`; // "Hello, John!"

// String methods
let text = "Hello";
console.log(text + " World"); // "Hello World"

4. Functions

What are Functions?

Functions are reusable blocks of code that perform a specific task. They help organize code, make it more readable, and avoid repetition.

Function Declaration

// Function declaration
function greet() {
    console.log("Hello, World!");
}

// Calling a function
greet(); // "Hello, World!"

// Function with parameters
function greet(name) {
    console.log(`Hello, ${name}!`);
}

greet("John"); // "Hello, John!"

// Function with return
function add(a, b) {
    return a + b;
}

let result = add(5, 3);
console.log(result); // 8

Function Expression

// Function expression
const greet = function(name) {
    console.log(`Hello, ${name}!`);
};

greet("John"); // "Hello, John!"

// Named function expression
const add = function sum(a, b) {
    return a + b;
};

Arrow Functions

// Arrow function (ES6)
const greet = (name) => {
    console.log(`Hello, ${name}!`);
};

// Single parameter - parentheses optional
const greet = name => {
    console.log(`Hello, ${name}!`);
};

// Single expression - return implicit
const add = (a, b) => a + b;

// No parameters
const sayHello = () => console.log("Hello!");

// Arrow function with object return
const createPerson = (name, age) => ({ name, age });

Default Parameters

// Default parameter values
function greet(name = "Guest") {
    console.log(`Hello, ${name}!`);
}

greet();        // "Hello, Guest!"
greet("John");  // "Hello, John!"

// Multiple default parameters
function createUser(name, age = 18, city = "Unknown") {
    return { name, age, city };
}

Rest Parameters

// Rest parameters (...)
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3));        // 6
console.log(sum(1, 2, 3, 4, 5));   // 15

// Rest with other parameters
function greet(greeting, ...names) {
    names.forEach(name => {
        console.log(`${greeting}, ${name}!`);
    });
}

greet("Hello", "John", "Jane", "Bob");

Higher-Order Functions

// Functions that take other functions as arguments
function processArray(arr, callback) {
    return arr.map(callback);
}

let numbers = [1, 2, 3, 4];
let doubled = processArray(numbers, x => x * 2);
console.log(doubled); // [2, 4, 6, 8]

// Functions that return functions
function multiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

let double = multiplier(2);
console.log(double(5)); // 10

Immediately Invoked Function Expression (IIFE)

// IIFE - executes immediately
(function() {
    console.log("This runs immediately!");
})();

// IIFE with parameters
(function(name) {
    console.log(`Hello, ${name}!`);
})("John");

// Arrow function IIFE
(() => {
    console.log("Arrow IIFE");
})();

Closures

// Closure - inner function has access to outer function's variables
function outer() {
    let count = 0;
    
    return function inner() {
        count++;
        return count;
    };
}

let counter = outer();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

5. Objects & Arrays

Objects

Objects are collections of key-value pairs. They are used to store and organize data.

Creating Objects

// Object literal
let person = {
    name: "John",
    age: 25,
    city: "New York"
};

// Using new Object()
let person2 = new Object();
person2.name = "Jane";
person2.age = 30;

// Object with methods
let person3 = {
    name: "Bob",
    age: 28,
    greet: function() {
        return `Hello, I'm ${this.name}`;
    }
};

// ES6 shorthand
let name = "Alice";
let age = 25;
let person4 = { name, age }; // { name: "Alice", age: 25 }

Accessing Object Properties

let person = {
    name: "John",
    age: 25
};

// Dot notation
console.log(person.name);  // "John"
console.log(person.age);   // 25

// Bracket notation
console.log(person["name"]); // "John"
console.log(person["age"]);  // 25

// Dynamic property access
let prop = "name";
console.log(person[prop]); // "John"

Object Methods

let person = {
    firstName: "John",
    lastName: "Doe",
    fullName: function() {
        return `${this.firstName} ${this.lastName}`;
    },
    // ES6 method shorthand
    greet() {
        return `Hello, ${this.firstName}!`;
    }
};

console.log(person.fullName()); // "John Doe"
console.log(person.greet());    // "Hello, John!"

Arrays

Arrays are ordered collections of values. They can contain any data type.

Creating Arrays

// Array literal
let fruits = ["apple", "banana", "cherry"];

// Using new Array()
let numbers = new Array(1, 2, 3);

// Empty array
let empty = [];

// Array with mixed types
let mixed = [1, "hello", true, { name: "John" }];

Array Methods

let fruits = ["apple", "banana", "cherry"];

// Adding elements
fruits.push("orange");        // Add to end
fruits.unshift("grape");      // Add to beginning

// Removing elements
fruits.pop();                 // Remove from end
fruits.shift();               // Remove from beginning

// Finding elements
fruits.indexOf("banana");      // 1
fruits.includes("apple");      // true

// Array iteration
fruits.forEach(fruit => {
    console.log(fruit);
});

// Map - creates new array
let lengths = fruits.map(fruit => fruit.length);

// Filter - creates new array
let longFruits = fruits.filter(fruit => fruit.length > 5);

// Reduce - reduces to single value
let sum = [1, 2, 3, 4].reduce((total, num) => total + num, 0);

// Find - finds first matching element
let found = fruits.find(fruit => fruit.length > 5);

// Some - checks if any element matches
let hasLong = fruits.some(fruit => fruit.length > 5);

// Every - checks if all elements match
let allLong = fruits.every(fruit => fruit.length > 5);

Array Destructuring

let fruits = ["apple", "banana", "cherry"];

// Basic destructuring
let [first, second, third] = fruits;
console.log(first);  // "apple"

// Skipping elements
let [first, , third] = fruits;

// Default values
let [first = "unknown", second] = fruits;

// Rest in destructuring
let [first, ...rest] = fruits;
console.log(rest); // ["banana", "cherry"]

Object Destructuring

let person = {
    name: "John",
    age: 25,
    city: "New York"
};

// Basic destructuring
let { name, age } = person;
console.log(name); // "John"

// Renaming
let { name: fullName, age: years } = person;

// Default values
let { name, age = 18 } = person;

// Rest in destructuring
let { name, ...rest } = person;

Spread Operator

// Array spread
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];
let combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6]

// Object spread
let obj1 = { a: 1, b: 2 };
let obj2 = { c: 3, d: 4 };
let merged = { ...obj1, ...obj2 }; // { a: 1, b: 2, c: 3, d: 4 }

// Copying
let arrCopy = [...arr1];
let objCopy = { ...obj1 };

6. DOM Manipulation

What is the DOM?

The Document Object Model (DOM) is a programming interface for HTML and XML documents. It represents the page structure as a tree of objects that can be manipulated with JavaScript.

Selecting Elements

// Select by ID
let element = document.getElementById("myId");

// Select by class (returns NodeList)
let elements = document.getElementsByClassName("myClass");

// Select by tag name (returns NodeList)
let paragraphs = document.getElementsByTagName("p");

// Query selector (returns first match)
let element = document.querySelector("#myId");
let element = document.querySelector(".myClass");
let element = document.querySelector("p");

// Query selector all (returns NodeList)
let elements = document.querySelectorAll(".myClass");
let elements = document.querySelectorAll("p");

Modifying Elements

// Change text content
element.textContent = "New text";
element.innerText = "New text";
element.innerHTML = "New text";

// Change attributes
element.setAttribute("id", "newId");
element.setAttribute("class", "newClass");
element.id = "newId";
element.className = "newClass";
element.classList.add("newClass");
element.classList.remove("oldClass");
element.classList.toggle("active");

// Change styles
element.style.color = "red";
element.style.backgroundColor = "blue";
element.style.fontSize = "20px";

Creating Elements

// Create new element
let newDiv = document.createElement("div");
newDiv.textContent = "New div";
newDiv.className = "container";

// Create text node
let textNode = document.createTextNode("Some text");

// Append to parent
let parent = document.getElementById("parent");
parent.appendChild(newDiv);

// Insert before
let sibling = document.getElementById("sibling");
parent.insertBefore(newDiv, sibling);

// Replace child
parent.replaceChild(newDiv, oldChild);

// Remove child
parent.removeChild(newDiv);

Event Listeners

// Add event listener
element.addEventListener("click", function() {
    console.log("Clicked!");
});

// Arrow function
element.addEventListener("click", () => {
    console.log("Clicked!");
});

// Named function
function handleClick() {
    console.log("Clicked!");
}
element.addEventListener("click", handleClick);

// Remove event listener
element.removeEventListener("click", handleClick);

// Event object
element.addEventListener("click", function(event) {
    console.log(event.target);
    console.log(event.type);
    event.preventDefault(); // Prevent default behavior
    event.stopPropagation(); // Stop event bubbling
});

Common Events

// Mouse events
element.addEventListener("click", handler);
element.addEventListener("dblclick", handler);

// Form events

// Window events

7. Events

What are Events?

Events are actions or occurrences that happen in the browser, such as clicks, key presses, form submissions, etc. JavaScript can listen for these events and execute code in response.

Event Types

Mouse Events

// Click event
button.addEventListener("click", function(event) {
    console.log("Button clicked!");
});

// Double click
element.addEventListener("dblclick", handler);

// Mouse enter/leave
element.addEventListener("mouseenter", handler);
element.addEventListener("mouseleave", handler);

// Mouse move
element.addEventListener("mousemove", handler);

// Mouse down/up
element.addEventListener("mousedown", handler);
element.addEventListener("mouseup", handler);

Keyboard Events

// Key down
input.addEventListener("keydown", function(event) {
    console.log("Key pressed:", event.key);
    console.log("Key code:", event.keyCode);
});

// Key up
input.addEventListener("keyup", handler);

// Key press
input.addEventListener("keypress", handler);

Form Events

// Submit
form.addEventListener("submit", function(event) {
    event.preventDefault(); // Prevent form submission
    // Handle form data
});

// Change
input.addEventListener("change", function(event) {
    console.log("Value changed:", event.target.value);
});

// Input (fires on every keystroke)
input.addEventListener("input", function(event) {
    console.log("Current value:", event.target.value);
});

// Focus/Blur
input.addEventListener("focus", handler);
input.addEventListener("blur", handler);

Event Object

element.addEventListener("click", function(event) {
    // Event properties
    console.log(event.type);        // "click"
    console.log(event.target);      // Element that triggered event
    console.log(event.currentTarget); // Element with listener
    console.log(event.clientX);     // Mouse X position
    console.log(event.clientY);     // Mouse Y position
    console.log(event.key);         // Key pressed
    console.log(event.preventDefault); // Function to prevent default
    console.log(event.stopPropagation); // Function to stop bubbling
});

Event Bubbling and Capturing

// Event bubbling (default) - event bubbles up from target
// Event capturing - event captured from top down

// Capturing phase
element.addEventListener("click", handler, true);

// Bubbling phase (default)
element.addEventListener("click", handler, false);
element.addEventListener("click", handler); // Same as false

// Stop propagation
element.addEventListener("click", function(event) {
    event.stopPropagation(); // Stops event from bubbling
});

Event Delegation

// Instead of adding listener to each element,
// add listener to parent and use event.target
let list = document.getElementById("list");

list.addEventListener("click", function(event) {
    if (event.target.tagName === "LI") {
        console.log("List item clicked:", event.target.textContent);
    }
});

8. Async JavaScript

What is Asynchronous JavaScript?

Asynchronous JavaScript allows code to run without blocking the main thread. This is essential for operations like fetching data, reading files, or waiting for user input.

Callbacks

// Callback function
function fetchData(callback) {
    setTimeout(() => {
        callback("Data received");
    }, 1000);
}

fetchData(function(data) {
    console.log(data); // "Data received" after 1 second
});

// Callback hell (nested callbacks)
getData(function(a) {
    getMoreData(a, function(b) {
        getEvenMoreData(b, function(c) {
            console.log(c);
        });
    });
});

Promises

// Creating a promise
let promise = new Promise(function(resolve, reject) {
    let success = true;
    
    if (success) {
        resolve("Success!");
    } else {
        reject("Error!");
    }
});

// Using promises
promise
    .then(function(result) {
        console.log(result); // "Success!"
    })
    .catch(function(error) {
        console.log(error); // "Error!"
    })
    .finally(function() {
        console.log("Always runs");
    });

// Promise chaining
fetchData()
    .then(result => processData(result))
    .then(processed => displayData(processed))
    .catch(error => console.error(error));

Async/Await

// Async function
async function fetchData() {
    try {
        let response = await fetch('https://api.example.com/data');
        let data = await response.json();
        return data;
    } catch (error) {
        console.error("Error:", error);
    }
}

// Using async/await
async function displayData() {
    let data = await fetchData();
    console.log(data);
}

displayData();

// Multiple async operations
async function fetchMultiple() {
    let [data1, data2, data3] = await Promise.all([
        fetchData1(),
        fetchData2(),
        fetchData3()
    ]);
    return { data1, data2, data3 };
}

Fetch API

// Basic fetch
fetch('https://api.example.com/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

// Fetch with async/await
async function getData() {
    try {
        let response = await fetch('https://api.example.com/data');
        let data = await response.json();
        console.log(data);
    } catch (error) {
        console.error("Error:", error);
    }
}

// POST request
fetch('https://api.example.com/data', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify({ name: "John", age: 25 })
})
    .then(response => response.json())
    .then(data => console.log(data));

setTimeout and setInterval

// setTimeout - executes once after delay
setTimeout(function() {
    console.log("This runs after 2 seconds");
}, 2000);

// setInterval - executes repeatedly
let interval = setInterval(function() {
    console.log("This runs every second");
}, 1000);

// Clear interval
clearInterval(interval);

// Clear timeout
let timeout = setTimeout(handler, 1000);
clearTimeout(timeout);
